26/// Makes a virtual page shared by revoking its validation, updating the
27/// page state, and modifying the page tables accordingly.
28///
29/// # Arguments
30///
31/// * `vaddr` - The virtual address of the page to be made shared.
32///
33/// # Safety
34///
35/// Converting the memory at `vaddr` must be safe within Rust's memory model.
36/// Notably any objects at `vaddr` must tolerate unsynchronized writes of any
37/// bit pattern.  In addition, the caller must take responsibility for
38/// returning a page to the private state if it is ever freed.
39pub unsafe fn make_page_shared(vaddr: VirtAddr) -> Result<(), SvsmError> {
40    // Revoke page validation before changing page state.
41    SVSM_PLATFORM.validate_virtual_page_range(
42        MemoryRegion::new(vaddr, PAGE_SIZE),
43        PageValidateOp::Invalidate,
44    )?;
45    let paddr = virt_to_phys(vaddr);
46    if valid_bitmap_valid_addr(paddr) {
47        valid_bitmap_clear_valid_4k(paddr);
48    }
49
50    // Ask the hypervisor to make the page shared.
51    SVSM_PLATFORM.page_state_change(
52        MemoryRegion::new(paddr, PAGE_SIZE),
53        PageSize::Regular,
54        PageStateChangeOp::Shared,
55    )?;
56
57    // Update the page tables to map the page as shared.
58    this_cpu()
59        .get_pgtable()
60        .set_shared_4k(vaddr)
61        .expect("Failed to remap shared page in page tables");
62    flush_tlb_global_sync();
63
64    Ok(())
65}

67/// Makes a virtual page private by updating the page tables, modifying the
68/// page state, and revalidating the page.
69///
70/// # Arguments
71///
72/// * `vaddr` - The virtual address of the page to be made private.
73///
74/// # Safety
75///
76/// Converting the memory at `vaddr` must be safe within Rust's memory model.
77/// No outstanding references to the page may exist.
78pub unsafe fn make_page_private(vaddr: VirtAddr) -> Result<(), SvsmError> {
79    // Update the page tables to map the page as private.
80    this_cpu().get_pgtable().set_encrypted_4k(vaddr)?;
81    flush_tlb_global_sync();
82
83    // Ask the hypervisor to make the page private.
84    let paddr = virt_to_phys(vaddr);
85    SVSM_PLATFORM.page_state_change(
86        MemoryRegion::new(paddr, PAGE_SIZE),
87        PageSize::Regular,
88        PageStateChangeOp::Private,
89    )?;
90
91    // Validate the page now that it is private again.
92    SVSM_PLATFORM.validate_virtual_page_range(
93        MemoryRegion::new(vaddr, PAGE_SIZE),
94        PageValidateOp::Validate,
95    )?;
96    if valid_bitmap_valid_addr(paddr) {
97        valid_bitmap_set_valid_4k(paddr);
98    }
99
100    Ok(())
101}

103/// SharedBox is a safe wrapper around memory pages shared with the host.
104pub struct SharedBox<T> {
105    ptr: NonNull<T>,
106}
107
108impl<T> SharedBox<T> {
109    /// Allocate some memory and share it with the host.
110    pub fn try_new_zeroed() -> Result<Self, SvsmError> {
111        let page_box = PageBox::<MaybeUninit<T>>::try_new_zeroed()?;
112        let vaddr = page_box.vaddr();
113
114        let ptr = NonNull::from(PageBox::leak(page_box)).cast::<T>();
115
116        for offset in (0..core::mem::size_of::<T>()).step_by(PAGE_SIZE) {
117            unsafe {
118                make_page_shared(vaddr + offset)?;
119            }
120        }
121
122        Ok(Self { ptr })
123    }
124
125    /// Returns the virtual address of the memory.
126    pub fn addr(&self) -> VirtAddr {
127        VirtAddr::from(self.ptr.as_ptr())
128    }
129
130    /// Read the currently stored value into `out`.
131    pub fn read_into(&self, out: &mut T)
132    where
133        T: FromBytes + Copy,
134    {
135        // SAFETY: `self.ptr` is valid. Any bitpattern is valid for `T`.
136        unsafe {
137            unsafe_copy_bytes(self.ptr.as_ptr(), out, 1);
138        }
139    }
140
141    /// Share `value` with the host.
142    pub fn write_from(&mut self, value: &T)
143    where
144        T: Copy,
145    {
146        // SAFETY: `self.ptr` is valid..
147        unsafe {
148            unsafe_copy_bytes(value, self.ptr.as_ptr(), 1);
149        }
150    }
151
152    /// Leak the memory.
153    pub fn leak(self) -> NonNull<T> {
154        let ptr = self.ptr;
155        core::mem::forget(self);
156        ptr
157    }
158}
159
160impl<T, const N: usize> SharedBox<[T; N]> {
161    /// Clear the first `n` elements.
162    pub fn nclear(&mut self, n: usize) -> Result<(), SvsmReqError>
163    where
164        T: FromZeros,
165    {
166        if n > N {
167            return Err(SvsmReqError::invalid_parameter());
168        }
169
170        // SAFETY: `self.ptr` is valid and we did a bounds check on `n`.
171        unsafe {
172            write_bytes(self.ptr.as_ptr().cast::<T>(), n, 0);
173        }
174
175        Ok(())
176    }
177
178    /// Fill up the `outbuf` slice provided with bytes from data
179    pub fn copy_to_slice(&self, outbuf: &mut [T]) -> Result<(), SvsmReqError>
180    where
181        T: FromBytes + Copy,
182    {
183        if outbuf.len() > N {
184            return Err(SvsmReqError::invalid_parameter());
185        }
186
187        // SAFETY: `self.ptr` is valid.
188        unsafe {
189            unsafe_copy_bytes(self.ptr.as_ptr().cast(), outbuf.as_mut_ptr(), outbuf.len());
190        }
191
192        Ok(())
193    }
194}

196unsafe impl<T> Send for SharedBox<T> where T: Send {}
197unsafe impl<T> Sync for SharedBox<T> where T: Sync {}
198
199impl<T> Drop for SharedBox<T> {
200    fn drop(&mut self) {
201        // Re-encrypt the pages.
202        let res = (0..size_of::<Self>())
203            .step_by(PAGE_SIZE)
204            .try_for_each(|offset| unsafe { make_page_private(self.addr() + offset) });
205
206        // If re-encrypting was successful free the memory otherwise leak it.
207        if res.is_ok() {
208            drop(unsafe { PageBox::from_raw(self.ptr.cast::<MaybeUninit<T>>()) });
209        } else {
210            log::error!("failed to set pages to encrypted. Memory leak!");
211        }
212    }
213}
